    <script>
        async function fetchServices() {
            try {
                const response = await fetch('https://localhost:7169/api/Service'); // URL مربوط به سرویس شما
                if (!response.ok) {
                    throw new Error('Network response was not ok');
                }
                const services = await response.json();
                console.log('Fetched services:', services);
                displayServices(services);
            } catch (error) {
                console.error('There has been a problem with your fetch operation:', error);
            }
        }

        function displayServices(services) {
            const serviceList = document.getElementById('serviceList');
            serviceList.innerHTML = '';
            services.forEach(service => {
                console.log('Service object:', service);
                const listItem = document.createElement('li');
                listItem.innerHTML = `
                    <div class="service-info">
                        <span class="service-title">Title: ${service.serviceTitle}</span>
                        <span class="service-number">Number: ${service.serviceNumber}</span>
                        <span class="service-date">Date: ${new Date(service.serviceDate).toLocaleDateString()}</span>
                        <span>Wage: ${service.wage}</span>
                        <span>Customer ID: ${service.customerId}</span>
                        <span>Car ID: ${service.carId}</span>
                        ${service.servicesDetails.length > 0 ? `<button class="details-button" onclick="toggleDetails(this)">Show Details</button>` : ''}
                        <div class="service-details">
                            ${service.servicesDetails.map(detail => `
                                <div>
                                    <span>Part ID: ${detail.partId}</span>
                                    <span>Service Type ID: ${detail.serviceTypeId}</span>
                                    <span>Wage: ${detail.wage}</span>
                                    <span>Part Price: ${detail.partPrice}</span>
                                </div>
                            `).join('')}
                        </div>
                    </div>
                `;
                console.log('List item:', listItem.innerHTML);
                serviceList.appendChild(listItem);
            });
        }

        function toggleDetails(button) {
            const details = button.nextElementSibling;
            if (details.style.display === "none" || details.style.display === "") {
                details.style.display = "block";
                button.textContent = "Hide Details";
            } else {
                details.style.display = "none";
                button.textContent = "Show Details";
            }
        }

        window.onload = () => {
            fetchServices();
        };
    </script>
